# 8. CDK 示例代码（TypeScript）

下面是一个非常标准的 CDK 版本。

## 8.1 Stack code

```ts
import * as cdk from 'aws-cdk-lib';
import { Construct } from 'constructs';
import * as s3 from 'aws-cdk-lib/aws-s3';
import * as lambda from 'aws-cdk-lib/aws-lambda';
import * as apigateway from 'aws-cdk-lib/aws-apigateway';

export class FileUploadStack extends cdk.Stack {
  constructor(scope: Construct, id: string, props?: cdk.StackProps) {
    super(scope, id, props);

    // Create S3 bucket
    const bucket = new s3.Bucket(this, 'UploadBucket', {
      versioned: true,
      removalPolicy: cdk.RemovalPolicy.DESTROY,
      autoDeleteObjects: true,
    });

    // Create Lambda function
    const uploadHandler = new lambda.Function(this, 'UploadHandler', {
      runtime: lambda.Runtime.NODEJS_18_X,
      handler: 'index.handler',
      code: lambda.Code.fromAsset('lambda'),
      environment: {
        BUCKET_NAME: bucket.bucketName,
      },
    });

    // Grant Lambda permission to access S3
    bucket.grantReadWrite(uploadHandler);

    // Create API Gateway
    const api = new apigateway.RestApi(this, 'UploadApi', {
      restApiName: 'File Upload Service',
    });

    const uploadResource = api.root.addResource('upload');
    uploadResource.addMethod('POST', new apigateway.LambdaIntegration(uploadHandler));
  }
}
```

---

## 8.2 Lambda code

```ts
import { S3 } from 'aws-sdk';

const s3 = new S3();

export const handler = async (event: any) => {
  try {
    const bucketName = process.env.BUCKET_NAME!;
    const body = JSON.parse(event.body || '{}');

    const fileName = body.fileName || `file-${Date.now()}.txt`;
    const content = body.content || 'default content';

    await s3.putObject({
      Bucket: bucketName,
      Key: fileName,
      Body: content,
    }).promise();

    return {
      statusCode: 200,
      body: JSON.stringify({
        message: 'File uploaded successfully',
        fileName: fileName,
      }),
    };
  } catch (error) {
    console.error('Upload failed:', error);

    return {
      statusCode: 500,
      body: JSON.stringify({
        message: 'Upload failed',
      }),
    };
  }
};
```

# 11. 更真实的升级版设计

真正生产环境里，不一定让客户端把文件内容先传给 Lambda。

更好的方式是：

```text
Client
  ↓
API Gateway
  ↓
Lambda
  ↓
Generate pre-signed URL
  ↓
Client uploads directly to S3
```

---

## 为什么更好

因为：

* 减少 Lambda 负担
* 减少 API payload 限制问题
* 更适合大文件
* 延迟更低
* 成本更低

---

## 面试回答

> For production-grade file upload, I would usually generate a pre-signed S3 URL in Lambda and let the client upload directly to S3. That avoids routing large payloads through Lambda and improves scalability and efficiency.

# 9. CDK 示例代码（TypeScript）

## `lib/resume-pipeline-stack.ts`

```typescript
import * as cdk from 'aws-cdk-lib';
import { Construct } from 'constructs';
import * as s3 from 'aws-cdk-lib/aws-s3';
import * as lambda from 'aws-cdk-lib/aws-lambda';
import * as s3n from 'aws-cdk-lib/aws-s3-notifications';
import * as sqs from 'aws-cdk-lib/aws-sqs';
import * as lambdaEventSources from 'aws-cdk-lib/aws-lambda-event-sources';
import * as dynamodb from 'aws-cdk-lib/aws-dynamodb';
import * as sns from 'aws-cdk-lib/aws-sns';

export class ResumePipelineStack extends cdk.Stack {
  constructor(scope: Construct, id: string, props?: cdk.StackProps) {
    super(scope, id, props);

    const bucket = new s3.Bucket(this, 'ResumeBucket');

    const queue = new sqs.Queue(this, 'ResumeQueue');

    const table = new dynamodb.Table(this, 'ResumeTable', {
      partitionKey: { name: 'fileName', type: dynamodb.AttributeType.STRING },
      billingMode: dynamodb.BillingMode.PAY_PER_REQUEST,
    });

    const topic = new sns.Topic(this, 'ResumeNotificationTopic');

    const metadataExtractor = new lambda.Function(this, 'MetadataExtractorFn', {
      runtime: lambda.Runtime.PYTHON_3_12,
      handler: 'app.lambda_handler',
      code: lambda.Code.fromAsset('lambda/metadata-extractor'),
      environment: {
        QUEUE_URL: queue.queueUrl,
      },
    });

    const resumeProcessor = new lambda.Function(this, 'ResumeProcessorFn', {
      runtime: lambda.Runtime.PYTHON_3_12,
      handler: 'app.lambda_handler',
      code: lambda.Code.fromAsset('lambda/resume-processor'),
      environment: {
        TABLE_NAME: table.tableName,
        TOPIC_ARN: topic.topicArn,
      },
    });

    bucket.addEventNotification(
      s3.EventType.OBJECT_CREATED,
      new s3n.LambdaDestination(metadataExtractor)
    );

    resumeProcessor.addEventSource(
      new lambdaEventSources.SqsEventSource(queue)
    );

    bucket.grantRead(metadataExtractor);
    queue.grantSendMessages(metadataExtractor);

    table.grantWriteData(resumeProcessor);
    topic.grantPublish(resumeProcessor);
    queue.grantConsumeMessages(resumeProcessor);
  }
}
```

```python
import json
import os
import boto3

sqs = boto3.client("sqs")
QUEUE_URL = os.environ["QUEUE_URL"]

def lambda_handler(event, context):
    for record in event["Records"]:
        bucket_name = record["s3"]["bucket"]["name"]
        file_name = record["s3"]["object"]["key"]
        file_size = record["s3"]["object"].get("size", 0)

        message = {
            "bucket": bucket_name,
            "file_name": file_name,
            "file_size": file_size
        }

        sqs.send_message(
            QueueUrl=QUEUE_URL,
            MessageBody=json.dumps(message)
        )

    return {
        "statusCode": 200,
        "body": "Metadata sent to SQS"
    }
```

## `lambda/resume-processor/app.py`

```python
import json
import os
import boto3
from datetime import datetime

dynamodb = boto3.resource("dynamodb")
sns = boto3.client("sns")

TABLE_NAME = os.environ["TABLE_NAME"]
TOPIC_ARN = os.environ["TOPIC_ARN"]

table = dynamodb.Table(TABLE_NAME)

def lambda_handler(event, context):
    for record in event["Records"]:
        body = json.loads(record["body"])

        file_name = body["file_name"]
        file_size = body["file_size"]
        bucket = body["bucket"]

        result = {
            "fileName": file_name,
            "bucket": bucket,
            "fileSize": file_size,
            "processedAt": datetime.utcnow().isoformat(),
            "status": "PROCESSED"
        }

        table.put_item(Item=result)

        sns.publish(
            TopicArn=TOPIC_ARN,
            Subject="Resume Processed",
            Message=f"Resume {file_name} has been processed successfully."
        )

    return {
        "statusCode": 200,
        "body": "Resume processed successfully"
    }
```

## `storage_database_stack.py`

```python
from aws_cdk import (
    Stack,
    RemovalPolicy,
    CfnOutput,
    SecretValue,
    aws_ec2 as ec2,
    aws_s3 as s3,
    aws_rds as rds,
)
from constructs import Construct


class StorageDatabaseStack(Stack):
    def __init__(self, scope: Construct, construct_id: str, **kwargs) -> None:
        super().__init__(scope, construct_id, **kwargs)

        # Create a VPC with public and private subnets.
        # max_azs=2 means the VPC can span across 2 availability zones.
        # nat_gateways=1 means private subnets can reach the internet for updates if needed.
        vpc = ec2.Vpc(
            self,
            "AppVpc",
            max_azs=2,
            nat_gateways=1,
        )

        # Create an S3 bucket.
        # versioned=True keeps old versions of files.
        # removal_policy=DESTROY is good for demo only, not production.
        # auto_delete_objects=True helps delete the bucket automatically when the stack is destroyed.
        bucket = s3.Bucket(
            self,
            "AppBucket",
            versioned=True,
            removal_policy=RemovalPolicy.DESTROY,
            auto_delete_objects=True,
        )

        # Create a security group for the RDS instance.
        # This controls which traffic can reach the database.
        db_security_group = ec2.SecurityGroup(
            self,
            "DatabaseSecurityGroup",
            vpc=vpc,
            description="Security group for RDS instance",
            allow_all_outbound=True,
        )

        # For demo purposes, allow MySQL access from anywhere.
        # In real production, you should restrict this to specific app servers or CIDR blocks.
        db_security_group.add_ingress_rule(
            ec2.Peer.any_ipv4(),
            ec2.Port.tcp(3306),
            "Allow MySQL access from anywhere for demo only",
        )

        # Create the RDS instance.
        # We use MySQL here.
        # The database is placed inside the VPC.
        # publicly_accessible=True is okay for demo, but production often uses private access only.
        db_instance = rds.DatabaseInstance(
            self,
            "AppDatabase",
            engine=rds.DatabaseInstanceEngine.mysql(
                version=rds.MysqlEngineVersion.VER_8_0_39
            ),
            instance_type=ec2.InstanceType.of(
                ec2.InstanceClass.BURSTABLE3,
                ec2.InstanceSize.MICRO,
            ),
            vpc=vpc,
            credentials=rds.Credentials.from_password(
                username="admin",
                password=SecretValue.unsafe_plain_text("DemoPassword123!")
            ),
            allocated_storage=20,
            max_allocated_storage=100,
            multi_az=False,
            publicly_accessible=True,
            security_groups=[db_security_group],
            database_name="appdb",
            removal_policy=RemovalPolicy.DESTROY,
            deletion_protection=False,
        )

        # Output the S3 bucket name so we can find it after deployment.
        CfnOutput(
            self,
            "BucketName",
            value=bucket.bucket_name,
            description="S3 bucket name",
        )

        # Output the RDS endpoint address.
        CfnOutput(
            self,
            "DatabaseEndpoint",
            value=db_instance.db_instance_endpoint_address,
            description="RDS endpoint address",
        )

        # Output the database port.
        CfnOutput(
            self,
            "DatabasePort",
            value=db_instance.db_instance_endpoint_port,
            description="RDS endpoint port",
        )
```

```java
package com.example.metadata;

import com.amazonaws.services.lambda.runtime.Context;
import com.amazonaws.services.lambda.runtime.RequestHandler;
import software.amazon.awssdk.services.sqs.SqsClient;
import software.amazon.awssdk.services.sqs.model.SendMessageRequest;
import org.json.JSONObject;

import java.util.Map;

public class MetadataExtractorHandler implements RequestHandler<Map<String, Object>, String> {

    private final SqsClient sqsClient = SqsClient.create();
    private final String queueUrl = System.getenv("QUEUE_URL");

    @Override
    public String handleRequest(Map<String, Object> event, Context context) {
        /*
         * In real production code, you would parse the S3 event object carefully.
         * Here we keep it simple for interview/demo purposes.
         */
        String bucketName = "resume-bucket";
        String fileName = "sample-resume.pdf";
        int fileSize = 1024;

        JSONObject message = new JSONObject();
        message.put("bucket", bucketName);
        message.put("fileName", fileName);
        message.put("fileSize", fileSize);
```

### `ResumeProcessorHandler.java`

```java
package com.example.processor;

import com.amazonaws.services.lambda.runtime.Context;
import com.amazonaws.services.lambda.runtime.RequestHandler;
import org.json.JSONObject;
import software.amazon.awssdk.services.dynamodb.DynamoDbClient;
import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
import software.amazon.awssdk.services.dynamodb.model.PutItemRequest;
import software.amazon.awssdk.services.sns.SnsClient;
import software.amazon.awssdk.services.sns.model.PublishRequest;

import java.time.Instant;
import java.util.HashMap;
import java.util.Map;

public class ResumeProcessorHandler implements RequestHandler<Map<String, Object>, String> {

    private final DynamoDbClient dynamoDbClient = DynamoDbClient.create();
    private final SnsClient snsClient = SnsClient.create();

    private final String tableName = System.getenv("TABLE_NAME");
    private final String topicArn = System.getenv("TOPIC_ARN");

    @Override
    public String handleRequest(Map<String, Object> event, Context context) {
        /*
         * In real production code, parse SQS event records.
         * Here we use sample data for demo/interview clarity.
         */
        String fileName = "sample-resume.pdf";
        String bucket = "resume-bucket";
        String fileSize = "1024";

        Map<String, AttributeValue> item = new HashMap<>();
        item.put("fileName", AttributeValue.builder().s(fileName).build());
        item.put("bucket", AttributeValue.builder().s(bucket).build());
        item.put("fileSize", AttributeValue.builder().n(fileSize).build());
        item.put("status", AttributeValue.builder().s("PROCESSED").build());
        item.put("processedAt", AttributeValue.builder().s(Instant.now().toString()).build());

        PutItemRequest putItemRequest = PutItemRequest.builder()
                .tableName(tableName)
                .item(item)
                .build();

        dynamoDbClient.putItem(putItemRequest);

        PublishRequest publishRequest = PublishRequest.builder()
                .topicArn(topicArn)
                .subject("Resume Processed")
                .message("Resume " + fileName + " has been processed successfully.")
                .build();

        snsClient.publish(publishRequest);

        return "Resume processed successfully";
    }
}
```
